thế có nghĩa là bỏ folder routes đi à
cấu trúc file của mình như này
backend/
│── node_modules/
│── routes/
│   ├── auth.js             # Xử lý login
│   ├── sensors.js          # API cho DataSensor + Home (lấy dữ liệu từ DB)
│   ├── actions.js          # API cho ActionHistory
│   └── profile.js          # API cho Profile
│
│── services/
│   ├── mqttService.js      # Kết nối & subscribe MQTT, lưu vào DB
│   └── db.js               # Kết nối database (MySQL/MongoDB)
│
│── models/
│   ├── Sensor.js           # Model bảng sensor_data
│   ├── Action.js           # Model bảng action_history
│   └── User.js             # Model bảng users (nếu cần cho login/profile)
│
│── app.js                  # Điểm khởi động Express (import routes + MQTT service)
│── .env                    # Config (DB, MQTT broker, PORT)
│── package.json


-- Bảng lưu thông tin người dùng
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


-- Bảng lưu dữ liệu cảm biến
CREATE TABLE sensor_data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    temperature DECIMAL(5,2),   -- ví dụ: 28.35
    humidity DECIMAL(5,2),      -- ví dụ: 65.20
    light INT,                  -- ví dụ: 320
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Bảng lưu lệnh điều khiển thiết bị
CREATE TABLE device_control (
    id INT AUTO_INCREMENT PRIMARY KEY,
    device_name VARCHAR(50) NOT NULL,  -- ví dụ: led, fan
    action VARCHAR(20) NOT NULL,       -- ví dụ: ON, OFF, SET
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
